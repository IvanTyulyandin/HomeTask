  Eu:ld 4 ; Euclid's alg

        br m2

        jmp Eu
    eCh:st 255 ; ds[4] - res of Euclid's alg
        ldc 1

        br nN ; nextNumber

        st 2

       ld 3

       st 3
       ld 3
       st 5
       ld 2
       st 4
